Un framework plus rapide que Jquery : Code source

Un code objet/héritage, sans Jquery, à la façon Java, réutilisable pour n'importe quel projet

Exemple de la base: 

var personne = ju.class(function() {
  this.constructor = function(nom, age) { }
} );

var etudiant = ju.extends(personne, function() {
 this.constructor = function(nom, age, numero) { }
} );

Cela m'a permis de comprendre le fondement des classes et leur utilité.
Tout le système repose sur les prototypes, avec le même système de constructeur que propose java.
De mon point de vue et expériences, il faut savoir utiliser les framework avec précaution.

Exemple de l'utilisation:

var player = ju.player({
   parent : "#playerId",
   attr : { "class" : "player", "volume" : "", "controls" :"controls"}
});

var control = ju.control({
   parent : "#controlId",
   attr : { "class" : "control" }
   random : {
     attr : { "width" : "20%" },
     on : { inner : "Random On" },
     off : { inner : "Random Off", attr : { "color" : "red" }}
   }
});

ju.attach(player, control);
